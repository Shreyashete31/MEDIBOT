Here's how to integrate this API with your frontend:

const API_BASE_URL = 'http://localhost:3001/api'; // Fetch remedies async function fetchRemedies() { try { const response = await fetch(`${API_BASE_URL}/remedies`); const data = await response.json(); if (data.success) { console.log('Remedies:', data.data); return data.data; } else { console.error('Error:', data.message); } } catch (error) { console.error('Network error:', error); } } // Search remedies with filters async function searchRemedies(searchTerm, category) { const params = new URLSearchParams(); if (searchTerm) params.append('search', searchTerm); if (category) params.append('category', category); const response = await fetch(`${API_BASE_URL}/remedies?${params}`); const data = await response.json(); return data.success ? data.data : []; }
